package tech;

import jakarta.servlet.ServletException;
import jakarta.servlet.annotation.WebServlet;
import jakarta.servlet.http.HttpServlet;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import java.io.IOException;

/**
 * Servlet implementation class tech
 */
@WebServlet("/tech")
public class tech extends HttpServlet {
    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws IOException {
        String text = request.getParameter("text");
        String findWord = request.getParameter("findWord");
        String replaceWord = request.getParameter("replaceWord");

        if (text == null || findWord == null || replaceWord == null) {
            response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Missing parameters");
            return;
        }

        // Simple replacement (case-insensitive)
        String updated = text.replaceAll("(?i)" + findWord, replaceWord);

        // Send JSON response
        response.setContentType("application/json");
        response.getWriter().write("{\"result\":\"" + updated.replace("\"","\\\"") + "\"}");
    }
}

    let originalText = document.getElementById("textInput").value;
    let lastResult = "";

    function doReplace() {
      const text = document.getElementById("textInput").value;
      const findWord = document.getElementById("findWord").value;
      const replaceWord = document.getElementById("replaceWord").value;

      if (!findWord) {
        alert("Please enter a word to find!");
        return;
      }

      // Replace all instances
      const regex = new RegExp(findWord, "gi");
      const updated = text.replace(regex, match => {
        // If it's a link, replace domain too
        if (match.toLowerCase().includes("http")) {
          return replaceWord.toLowerCase();
        }
        return replaceWord;
      });

      lastResult = document.getElementById("result").innerHTML;
      document.getElementById("result").innerHTML = updated.replace(regex, `<span class="highlight">${replaceWord}</span>`);
    }

    function undoReplace() {
      if (lastResult) {
        document.getElementById("result").innerHTML = lastResult;
      } else {
        document.getElementById("result").innerHTML = originalText;
      }
    }

    // Show initial text
    document.getElementById("result").textContent = originalText;
  </script>

function doReplace() {
  const text = document.getElementById("textInput").value;
  const findWord = document.getElementById("findWord").value;
  const replaceWord = document.getElementById("replaceWord").value;

  fetch("http://localhost:8080/SmartReplace/replace", {
    method: "POST",
    headers: { "Content-Type": "application/x-www-form-urlencoded" },
    body: `text=${encodeURIComponent(text)}&findWord=${findWord}&replaceWord=${replaceWord}`
  })
  .then(res => res.json())
  .then(data => {
    document.getElementById("result").innerText = data.result;
  })
  .catch(err => console.error(err));
}
